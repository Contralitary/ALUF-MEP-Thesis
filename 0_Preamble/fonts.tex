\usepackage{iftex}
    \ifPDFTeX    % If the pdfLaTeX-Compiler is used
        \usepackage[utf8]{inputenc}
        \usepackage[T1]{fontenc}
        \usepackage{mathptmx}    % Set main font equivalent to Times
        \usepackage[letterspace=70]{microtype}
        \usepackage{anyfontsize}
        %\addtokomafont{disposition}{\fontfamily{arial}}
        % If you would like the Arial font
            \newcommand*{\myfont}{\fontfamily{phv}\selectfont}
            % Set local font (\myfont) for title page equivalent to Arial
    \else    % If the XeLaTeX/LuaLaTeX-Compiler is used
        \usepackage{fontspec}
            \setmainfont{Times New Roman}
            \newfontfamily\myfont[Ligatures=TeX]{Arial}
            % Set local font (\myfont): Arial for title page
            %\newenvironment{ArialFont}{\myfont}{\par}
    
            \newfontfamily\tgtermes{texgyretermes}[
                Path = Fonts/TeX Gyre Termes/ ,
                Extension      = .otf ,
                UprightFont    = *-regular ,
                ItalicFont     = *-italic ,
                BoldFont       = *-bold ,
                BoldItalicFont = *-bolditalic ,
                ]
        \makeatletter
            \begingroup
                \tgtermes
                \DeclareFontShape{\f@encoding}{\rmdefault}{m}{sc}{%
                    <-> ssub * \f@family/m/sc}{}
                \DeclareFontShape{\f@encoding}{\rmdefault}{bx}{sc}{%
                    <-> ssub * \f@family/bx/sc}{}
            \endgroup
        \makeatother
    % Since Times New Roman does not support small caps, I use the above Termes font as an alternative
    \fi

\setkomafont{chapter}{\normalfont\bfseries\huge}
\setkomafont{section}{\normalfont\bfseries\Large}
